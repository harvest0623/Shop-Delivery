### 2.1 数据库配置

**位置**：`backend/config/database.js`

使用 **MySQL2** 连接池管理数据库连接：
```javascript
const pool = mysql.createPool({
    host: 'localhost',
    user: 'root',
    password: '123456',

    connectionLimit: 10,
```

**答辩要点**：使用连接池而不是单连接，可以管理多个并发请求，提高性能。

### 2.3 视图（4个）

#### 1) order_detail_view（订单详情视图）
**功能**：整合订单、顾客、商家、地址信息，方便查看订单完整详情

```sql
CREATE VIEW order_detail_view AS
SELECT 
    o.id AS order_id, o.order_no, o.total_amount, o.delivery_fee,
    o.status, o.remark, o.return_reason, o.created_at,
    c.username, c.phone AS customer_phone, c.nickname,
    s.name AS shop_name, s.image_url AS shop_image,
    CONCAT(a.province, a.city, a.district, a.detail) AS address_detail
FROM orders o
JOIN customers c ON o.customer_id = c.id
JOIN shops s ON o.shop_id = s.id
LEFT JOIN addresses a ON o.address_id = a.id;
```

**使用场景**：订单列表页面展示订单时，一次性获取所有关联信息

#### 2) shop_product_view（商家商品视图）

```sql
CREATE VIEW shop_product_view AS
```

#### 3) sales_summary_view（销售汇总视图）
**功能**：按商家汇总销售数据

```sql
CREATE VIEW sales_summary_view AS
SELECT 
    s.id AS shop_id, s.name AS shop_name,
    COUNT(DISTINCT o.id) AS total_orders,
    SUM(CASE WHEN o.status = 'completed' THEN o.total_amount ELSE 0 END) AS total_sales,
    SUM(CASE WHEN o.status = 'returned' THEN o.total_amount ELSE 0 END) AS total_returns
FROM shops s
LEFT JOIN orders o ON s.id = o.shop_id
GROUP BY s.id, s.name;
```

#### 4) customer_order_view（顾客订单视图）
**功能**：按顾客汇总订单数据

```sql
CREATE VIEW customer_order_view AS

    COUNT(o.id) AS total_orders,
    SUM(CASE WHEN o.status = 'completed' THEN o.total_amount ELSE 0 END) AS total_spent

LEFT JOIN orders o ON c.id = o.customer_id
GROUP BY c.id, c.username, c.nickname;
```

**答辩要点**：视图是虚拟表，不存储数据，查询时动态关联多表，简化复杂查询。

### 2.4 存储过程（4个）

#### 1) CreateOrder（创建订单）
**功能**：创建新订单，自动生成订单编号

```sql
DELIMITER //
CREATE PROCEDURE CreateOrder(
    IN p_customer_id INT,
    IN p_shop_id INT,
    IN p_address_id INT,
    IN p_total_amount DECIMAL(10,2),
    IN p_delivery_fee DECIMAL(5,2),
    IN p_remark TEXT
)
BEGIN
    DECLARE v_order_no VARCHAR(50);
    
    -- 自动生成订单编号：ORD + 年月日 + 4位随机数
    SET v_order_no = CONCAT('ORD', DATE_FORMAT(NOW(), '%Y%m%d'), 
                            LPAD(FLOOR(RAND() * 10000), 4, '0'));
    
    INSERT INTO orders (customer_id, shop_id, address_id, total_amount, 
                        delivery_fee, status, order_no, remark)
    VALUES (p_customer_id, p_shop_id, p_address_id, p_total_amount, 
            p_delivery_fee, 'pending', v_order_no, p_remark);
    
    SELECT LAST_INSERT_ID() AS order_id, v_order_no AS order_no;
END //
DELIMITER ;
```

**使用位置**：`backend/routes/orders.js` 的下单接口

#### 2) ProcessReturn（处理退货）
**功能**：处理订单退货申请，验证订单状态

```sql
DELIMITER //
CREATE PROCEDURE ProcessReturn(
    IN p_order_id INT,
    IN p_reason VARCHAR(255)
)
BEGIN
    DECLARE v_status VARCHAR(20);
    
    SELECT status INTO v_status FROM orders WHERE id = p_order_id;
    
    IF v_status = 'completed' THEN
        UPDATE orders 
        SET status = 'returned', return_reason = p_reason
        WHERE id = p_order_id;
        SELECT 'Return Success' AS message;
    ELSE
        SELECT 'Order status does not allow return' AS message;
    END IF;
END //
DELIMITER ;
```

#### 3) GetSalesStatistics（获取销售统计）
**功能**：根据日期范围查询订单统计数据

```sql
DELIMITER //
CREATE PROCEDURE GetSalesStatistics(
    IN p_start_date DATE,
    IN p_end_date DATE
)
BEGIN
    SELECT 
        COUNT(*) AS total_orders,
        SUM(total_amount) AS total_amount,
        COUNT(CASE WHEN status = 'completed' THEN 1 END) AS completed_count,
        COUNT(CASE WHEN status = 'returned' THEN 1 END) AS returned_count
    FROM orders
    WHERE DATE(created_at) BETWEEN p_start_date AND p_end_date;
END //
DELIMITER ;
```

#### 4) AddToCart（添加购物车）
**功能**：添加商品到购物车，检查库存，已存在则增加数量

```sql
DELIMITER //
CREATE PROCEDURE AddToCart(
    IN p_customer_id INT,
    IN p_product_id INT,
    IN p_quantity INT
)
BEGIN
    DECLARE v_stock INT;
    DECLARE v_exists INT;
    
    SELECT stock INTO v_stock FROM products WHERE id = p_product_id;
    
    IF v_stock < p_quantity THEN
        SELECT 'Insufficient stock' AS message;
    ELSE
        SELECT COUNT(*) INTO v_exists FROM cart 
        WHERE customer_id = p_customer_id AND product_id = p_product_id;
        
        IF v_exists > 0 THEN
            UPDATE cart SET quantity = quantity + p_quantity
            WHERE customer_id = p_customer_id AND product_id = p_product_id;
        ELSE
            INSERT INTO cart (customer_id, product_id, quantity)
            VALUES (p_customer_id, p_product_id, p_quantity);
        END IF;
        SELECT 'Added successfully' AS message;
    END IF;
END //
DELIMITER ;
```

**答辩要点**：存储过程预编译在数据库中，减少网络传输，提高执行效率，可以封装业务逻辑。

### 2.5 触发器（3个）

#### 1) trg_after_order_insert（订单创建减库存）
**功能**：当订单项被插入时，自动减少商品库存

```sql
DELIMITER //
CREATE TRIGGER trg_after_order_insert
AFTER INSERT ON order_items
FOR EACH ROW
BEGIN
    UPDATE products 
    SET stock = stock - NEW.quantity
    WHERE id = NEW.product_id;
END //
DELIMITER ;
```

**使用场景**：用户下单后，自动扣减商品库存

#### 2) trg_after_order_return（订单退货恢复库存）
**功能**：订单状态更新为退货时，自动恢复库存

```sql
DELIMITER //
CREATE TRIGGER trg_after_order_return
AFTER UPDATE ON orders
FOR EACH ROW
BEGIN
    IF OLD.status != 'returned' AND NEW.status = 'returned' THEN
        UPDATE products p
        JOIN order_items oi ON p.id = oi.product_id
        SET p.stock = p.stock + oi.quantity
        WHERE oi.order_id = NEW.id;
    END IF;
END //
DELIMITER ;
```

#### 3) trg_after_order_cancel（订单取消恢复库存）
**功能**：订单状态更新为取消时，自动恢复库存

```sql
DELIMITER //
CREATE TRIGGER trg_after_order_cancel
AFTER UPDATE ON orders
FOR EACH ROW
BEGIN
    IF OLD.status != 'cancelled' AND NEW.status = 'cancelled' THEN
        UPDATE products p
        JOIN order_items oi ON p.id = oi.product_id
        SET p.stock = p.stock + oi.quantity
        WHERE oi.order_id = NEW.id;
    END IF;
END //
DELIMITER ;
```

**答辩要点**：触发器是自动执行的存储过程，当特定事件（INSERT/UPDATE/DELETE）发生时自动触发，保证数据一致性。

### 3.3 核心API实现示例

#### 下单接口（orders.js）
```javascript
router.post('/', async (req, res) => {
    const { customer_id, shop_id, address_id, items, total_amount, delivery_fee, remark } = req.body;
    
    // 1. 创建订单
    const [orderResult] = await pool.query(
        'INSERT INTO orders (customer_id, shop_id, address_id, total_amount, delivery_fee, status, order_no, remark) VALUES (?, ?, ?, ?, ?, ?, ?, ?)',
        [customer_id, shop_id, address_id, total_amount, delivery_fee, 'pending', generateOrderNo(), remark]
    );
    
    const orderId = orderResult.insertId;
    
    // 2. 创建订单项（触发器自动减库存）
    for (const item of items) {
        await pool.query(
            'INSERT INTO order_items (order_id, product_id, quantity, price) VALUES (?, ?, ?, ?)',
            [orderId, item.id, item.quantity, item.price]
        );
    }
    
    // 3. 清空购物车
    await pool.query('DELETE FROM cart WHERE customer_id = ?', [customer_id]);
    
    res.json({ success: true, order_id: orderId, order_no: generateOrderNo() });
});
```

#### 退货接口（orders.js）
```javascript
router.put('/:id/return', async (req, res) => {
```

## 六、答辩要点总结

### 6.1 数据库设计亮点
1. **第三范式**：所有表满足3NF，消除数据冗余
2. **外键约束**：保证数据参照完整性
3. **视图**：简化多表查询，提高开发效率
4. **存储过程**：封装业务逻辑，减少网络传输
5. **触发器**：自动维护库存一致性

### 6.2 功能亮点
1. **完整的订单流程**：浏览→加购→下单→配送→完成→退货
2. **实时库存管理**：下单自动减库存，退货自动恢复
3. **多维度统计**：订单、商品、商家、趋势分析
4. **响应式设计**：适配PC和移动端
5. **用户体验**：动画效果、加载状态、空状态提示

### 6.3 技术亮点
1. **前后端分离**：Vue3 + Express + MySQL
2. **RESTful API**：统一的接口设计规范
3. **本地存储**：购物车数据持久化
4. **组件化开发**：可复用的UI组件
